396  bool neg = d.x < 0;
397  T dx;
398 
399  if (d.x == 0) {
400 #if DUAL_DISCONTINUITY_NAN

402 #elif DUAL_DISCONTINUITY_AVERAGE
403  dx = 0;
404 #elif DUAL_DISCONTINUITY_LEFT
405  dx = d.dx > 0 ? -d.dx : d.dx;

444  T x = lt ? d1.x : d2.x;
445  T dx;
446  if (d1.x == d2.x) {
447 #if DUAL_DISCONTINUITY_NAN

449 #elif DUAL_DISCONTINUITY_AVERAGE

453 #elif DUAL_DISCONTINUITY_RIGHT
454  dx = (d1.dx > d2.dx) ? d1.dx : d2.dx;
