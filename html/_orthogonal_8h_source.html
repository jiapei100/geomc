48  template <typename T, index_t N>
49  Vec<T,N> orthogonal(const Vec<T,N> v[N-1]) {
50  Vec<T,N> o;
51  o[N-1] = 1;
52  index_t P[N];
53  T m[N * (N-1)];
54  std::copy(v[0].begin(), v[0].begin() + N * (N-1), m);
55  bool parity_swap = false;
56 
57  if (decompLUP(m, N-1, N, P, &parity_swap)) {
58  // matrix is singular; nullity is > 1. return 0 vector.
59  return Vec<T,N>();
60  }
61 
62  for (index_t r = N-2; r >= 0; r--) {
63  // back substitute.
64  for (index_t c = N-1; c > r; c--) {
65  o[r] -= o[c] * _MxElem(m,r,c);
66  }
67  o[r] /= _MxElem(m,r,r);
68  }
69 
70  PermutationMatrix<N> Pmtx;
71  Pmtx.setRowSources(P);
72 
73  // invert the permutation.
74  // TODO: is that parity_swap correct?
75  return (parity_swap ? -1 : 1) * o * Pmtx;
76  }
