172  const Vec<T,3> rotate(T u, T v, T w, T radians) const {
173  T u2 = u*u;
174  T v2 = v*v;
175  T w2 = w*w;
176  T uxvywz = u*detail::VecBase<T,3>::x+v*detail::VecBase<T,3>::y+w*detail::VecBase<T,3>::z;
177  T uvw2 = u2+v2+w2;
178  T rsin = sqrt(uvw2)*sin(radians);
179  T c = cos(radians);
180 
181  return Vec<T,3>(
182  (u*uxvywz+(detail::VecBase<T,3>::x*(v2 + w2)-u*(v*detail::VecBase<T,3>::y+w*detail::VecBase<T,3>::z))*c+rsin*(-w*detail::VecBase<T,3>::y+v*detail::VecBase<T,3>::z))/uvw2,
183  (v*uxvywz+(detail::VecBase<T,3>::y*(u2 + w2)-v*(u*detail::VecBase<T,3>::x+w*detail::VecBase<T,3>::z))*c+rsin*( w*detail::VecBase<T,3>::x-u*detail::VecBase<T,3>::z))/uvw2,
184  (w*uxvywz+(detail::VecBase<T,3>::z*(u2 + v2)-w*(u*detail::VecBase<T,3>::x+v*detail::VecBase<T,3>::y))*c+rsin*(-v*detail::VecBase<T,3>::x+u*detail::VecBase<T,3>::y))/uvw2);
185  }

191  const Vec<T,3> rotate(T u, T v, T w, T a, T b, T c, T radians){
192  //rotation about arbitrary axis <u,v,w> and point <a,b,c>
193  T x = detail::VecBase<T,3>::x;
194  T y = detail::VecBase<T,3>::y;
195  T z = detail::VecBase<T,3>::z;
196 
197  T cc = cos(radians);
198  T ss = sin(radians);
199  T oneMcos = 1 - cc;
200 
201  T u2 = u*u;
202  T v2 = v*v;
203  T w2 = w*w;
204 
205  T uxvywz = u*x - v*y - w*z;
206 
207  return Vec<T,3>(
208  (a*(v2 + w2) - u*(b*v + c*w - uxvywz))*oneMcos + x*cc + (-c*v + b*w - w*y + v*z)*ss,
209  (b*(u2 + w2) - v*(a*u + c*w - uxvywz))*oneMcos + y*cc + ( c*u - a*w + w*x - u*z)*ss,
210  (c*(u2 + v2) - w*(a*u + b*v - uxvywz))*oneMcos + z*cc + (-b*u + a*v - v*x + u*y)*ss);
211  }
