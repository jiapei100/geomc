221  Vec<T,3> rotate(T u, T v, T w, T radians) const {
222  T u2 = u*u;
223  T v2 = v*v;
224  T w2 = w*w;
225  T uxvywz = u*detail::VecBase<T,3>::x+v*detail::VecBase<T,3>::y+w*detail::VecBase<T,3>::z;
226  T uvw2 = u2+v2+w2;
227  T rsin = std::sqrt(uvw2)*std::sin(radians);
228  T c = std::cos(radians);
229 
230  return Vec<T,3>(
231  (u*uxvywz+(detail::VecBase<T,3>::x*(v2 + w2)-u*(v*detail::VecBase<T,3>::y+w*detail::VecBase<T,3>::z))*c+rsin*(-w*detail::VecBase<T,3>::y+v*detail::VecBase<T,3>::z))/uvw2,
232  (v*uxvywz+(detail::VecBase<T,3>::y*(u2 + w2)-v*(u*detail::VecBase<T,3>::x+w*detail::VecBase<T,3>::z))*c+rsin*( w*detail::VecBase<T,3>::x-u*detail::VecBase<T,3>::z))/uvw2,
233  (w*uxvywz+(detail::VecBase<T,3>::z*(u2 + v2)-w*(u*detail::VecBase<T,3>::x+v*detail::VecBase<T,3>::y))*c+rsin*(-v*detail::VecBase<T,3>::x+u*detail::VecBase<T,3>::y))/uvw2);
234  }

247  Vec<T,3> rotate(T u, T v, T w, T a, T b, T c, T radians){
248  T x = detail::VecBase<T,3>::x;
249  T y = detail::VecBase<T,3>::y;
250  T z = detail::VecBase<T,3>::z;
251 
252  T cc = std::cos(radians);
253  T ss = std::sin(radians);
254  T oneMcos = 1 - cc;
255 
256  T u2 = u*u;
257  T v2 = v*v;
258  T w2 = w*w;
259 
260  T uxvywz = u*x - v*y - w*z;
261 
262  return Vec<T,3>(
263  (a*(v2 + w2) - u*(b*v + c*w - uxvywz))*oneMcos + x*cc + (-c*v + b*w - w*y + v*z)*ss,
264  (b*(u2 + w2) - v*(a*u + c*w - uxvywz))*oneMcos + y*cc + ( c*u - a*w + w*x - u*z)*ss,
265  (c*(u2 + v2) - w*(a*u + b*v - uxvywz))*oneMcos + z*cc + (-b*u + a*v - v*x + u*y)*ss);
266  }
