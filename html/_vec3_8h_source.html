226  const Vec<T,3> rotate(T u, T v, T w, T radians) const {
227  T u2 = u*u;
228  T v2 = v*v;
229  T w2 = w*w;
230  T uxvywz = u*detail::VecBase<T,3>::x+v*detail::VecBase<T,3>::y+w*detail::VecBase<T,3>::z;
231  T uvw2 = u2+v2+w2;
232  T rsin = sqrt(uvw2)*sin(radians);
233  T c = cos(radians);
234 
235  return Vec<T,3>(
236  (u*uxvywz+(detail::VecBase<T,3>::x*(v2 + w2)-u*(v*detail::VecBase<T,3>::y+w*detail::VecBase<T,3>::z))*c+rsin*(-w*detail::VecBase<T,3>::y+v*detail::VecBase<T,3>::z))/uvw2,
237  (v*uxvywz+(detail::VecBase<T,3>::y*(u2 + w2)-v*(u*detail::VecBase<T,3>::x+w*detail::VecBase<T,3>::z))*c+rsin*( w*detail::VecBase<T,3>::x-u*detail::VecBase<T,3>::z))/uvw2,
238  (w*uxvywz+(detail::VecBase<T,3>::z*(u2 + v2)-w*(u*detail::VecBase<T,3>::x+v*detail::VecBase<T,3>::y))*c+rsin*(-v*detail::VecBase<T,3>::x+u*detail::VecBase<T,3>::y))/uvw2);
239  }

252  const Vec<T,3> rotate(T u, T v, T w, T a, T b, T c, T radians){
253  T x = detail::VecBase<T,3>::x;
254  T y = detail::VecBase<T,3>::y;
255  T z = detail::VecBase<T,3>::z;
256 
257  T cc = cos(radians);
258  T ss = sin(radians);
259  T oneMcos = 1 - cc;
260 
261  T u2 = u*u;
262  T v2 = v*v;
263  T w2 = w*w;
264 
265  T uxvywz = u*x - v*y - w*z;
266 
267  return Vec<T,3>(
268  (a*(v2 + w2) - u*(b*v + c*w - uxvywz))*oneMcos + x*cc + (-c*v + b*w - w*y + v*z)*ss,
269  (b*(u2 + w2) - v*(a*u + c*w - uxvywz))*oneMcos + y*cc + ( c*u - a*w + w*x - u*z)*ss,
270  (c*(u2 + v2) - w*(a*u + b*v - uxvywz))*oneMcos + z*cc + (-b*u + a*v - v*x + u*y)*ss);
271  }
