41  Hit<T,N> trace(const Ray<T,N> &ray, HitSide side) const {
42  Hit<T,N> h = Hit<T,N>(ray, side); // defaults to miss
43  T r2 = r*r;
44  // if inside, we are guaranteed to hit the back. return miss if not tracing backface.
45  if ((ray.origin.dist2(center) < r2) and !(side & HIT_BACK)) return h;
46  const Vec<T,N> &dir = ray.direction; //for shorthand
47  Vec<T,N> x0 = center - ray.origin;
48  // solve for s such that ||s*ray - ctr|| == radius
49  T a = dir.dot(dir);
50  T b = -2*dir.dot(x0);
51  T c = x0.dot(x0) - r2;
52  T roots[2];
53  if (quadratic_solve(roots,a,b,c)){
54  T s, s0, s1;
55  // order the roots along the ray, from -inf; s0 first.
56  if (roots[0] < roots[1]){
57  s0 = roots[0];
58  s1 = roots[1];
59  } else {
60  s0 = roots[1];
61  s1 = roots[0];
62  }
63  // lesser root will always be the frontside, greater will be the back.
64  // if tracing both front and back and both hit, return the nearer (front).
65  if (s0 > 0 && (side & HIT_FRONT)){
66  s = s0;
67  h.side = HIT_FRONT;
68  } else if (s1 > 0 && (side & HIT_BACK)){
69  s = s1;
70  h.side = HIT_BACK;
71  } else {
72  return h; // hit behind origin; return miss.
73  }
74 
75  // successful hit
76  h.p = ray.atMultiple(s);
77  h.n = (h.p - center).unit();
78  h.s = s;
79  h.hit = true;
80  return h;
81  } else {
82  // no intersection; return miss.
83  return h;
84  }
85  }
