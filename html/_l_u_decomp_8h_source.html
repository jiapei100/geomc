35 index_t decompLUP(T* m, index_t rows, index_t cols, index_t *reorder, bool *swap_parity) {
36  const index_t n = std::min(rows, cols);
37  index_t degenerate_ct = 0;
38  *swap_parity = false;
39  // fill permutation array
40  for (index_t i = 0; i < cols; i++){
41  reorder[i] = i;
42  }
43 
44  for (index_t i = 0; i < n - 1; i++) {
45  // find pivot
46  T biggest = std::abs(_MxElem(i,i));
47  index_t pvt = i;
48  for (index_t p = i + 1; p < cols; p++) {
49  T pvt_val = std::abs(_MxElem(i,p));
50  if (pvt_val > biggest) {
51  pvt = p;
52  biggest = pvt_val;
53  }
54  }
55 
56  if (biggest == 0) {
57  // singular matrix
58  // could test against an epsilon to
59  // find ill-conditioned matrices
60  degenerate_ct += 1;
61  continue;
62  } else if (pvt != i) {
63  // swap col <i> with col <pvt>
64  for (index_t r = 0; r < rows; r++) {
65  std::swap(_MxElem(r, i),
66  _MxElem(r, pvt));
67  }
68  // make a note of the permutation in <P>
69  std::swap(reorder[i], reorder[pvt]);
70  *swap_parity = !*swap_parity;
71  }
72 
73  // eliminate lower elements
74  T a = _MxElem(i,i);
75  for (index_t r = i + 1; r < rows; r++) {
76  T b = _MxElem(r,i) / a;
77  for (index_t c = i + 1; c < cols; c++) {
78  // R_r = R_r - b * R_i
79  T src_elem = _MxElem(i,c);
80  T dst_elem = _MxElem(r,c);
81  _MxElem(r,c) = dst_elem - b * src_elem;
82  }
83  // set the lower matrix
84  _MxElem(r,i) = b;
85  }
86  }
87 
88  return degenerate_ct;
89 }

487  template <typename S>
488  void _linearSolve(S *dest, const S *b, bool permute=true) const {
489  const index_t n = LU.rows();
490  const index_t *p = P.getRowSources(); // permutation map
491 
492  // <y> and <dest>'s elements are used such that
493  // y[i] is never read after dest[i] is written.
494  // thus we may collapse their storage and save space:
495  S *y = dest;
496 
497  // LUx = Pb
498  // (Ux) is a vector, so let's solve for it:
499  // Ly = Pb
500  y[0] = permute ? b[p[0]] : b[0];
501  for (index_t r = 1; r < n; r++) {
502  y[r] = permute ? b[p[r]] : b[r];
503  for (index_t c = 0; c < r; c++) {
504  y[r] -= y[c] * LU.get(r,c);
505  }
506  }
507  // now with y, we may obtain x from:
508  // Ux = y
509  for (index_t r = n - 1; r >= 0; r--) {
510  // dest[r] = y[r]; // nop; dest and y are the same!
511  for (index_t c = n - 1; c > r; c--) {
512  dest[r] -= dest[c] * LU.get(r,c);
513  }
514  dest[r] /= LU.get(r,r);
515  }
516  }

520  void _inverseTranspose(S *dest) const {
521  // Set LUx = PI and solve for x, choosing columns of PI one at a time.
522  // Here, we use the rows of our destination matrix as though they are
523  // column vectors, and the caller will transpose.
524  const index_t *p = P.getColSources();

528  dest[p[i]] = 1;
529  _linearSolve(dest, dest, false);
