218  template <typename T>
219  void rotmat(SimpleMatrix<T,4,4> *into, T x, T y, T z, T theta) {
220  T c = cos(theta);
221  T s = sin(theta);
222  T oneMcos = 1 - c;
223  T m[4][4] = {
224  {c + oneMcos*x*x, oneMcos*x*y - s*z, oneMcos*x*z + s*y, 0},
225  {oneMcos*y*x + s*z, c + oneMcos*y*y, oneMcos*y*z - s*x, 0},
226  {oneMcos*z*x - s*y, oneMcos*z*y + s*x, c + oneMcos*z*z, 0},
227  {0, 0, 0, 1}
228  };
229  T *from = m[0];
230  std::copy(from, from+16, into->begin());
231  }

240  template <typename T>
241  void rotmat(SimpleMatrix<T,4,4> *into, const Quat<T> &q) {
242  T x = q.x; //for convenience
243  T y = q.y;
244  T z = q.z;
245  T w = q.w;
246  T x2 = x*x; // for speed
247  T y2 = y*y; // ...
248  T z2 = z*z;
249  T w2 = w*w;
250  T twox = 2*x;
251  T twoy = 2*y;
252  T twoz = 2*z;
253  T xy = twox * y;
254  T xz = twox * z;
255  T xw = twox * w;
256  T yz = twoy * z;
257  T yw = twoy * w;
258  T zw = twoz * w;
259  T m[4][4] = {
260  {w2 + x2 - y2 - z2, xy - zw, xz + yw, 0},
261  {xy + zw, w2 - x2 + y2 - z2, yz - xw, 0},
262  {xz - yw, yz + xw, w2 - x2 - y2 + z2, 0},
263  {0, 0, 0, 1}
264  };
265  T *from = m[0];
266  std::copy(from, from+16, into->begin());
267  }

271  void rotmat(SimpleMatrix<T,4,4> *into,
272  T x, T y, T z,
273  T px, T py, T pz,

276  T s = sin(theta);
277  T oneMcos = 1 - c;
278  T x2 = x*x;
279  T y2 = y*y;
280  T z2 = z*z;
281  T m[4][4] = {
282  {x2 + c*(y2 + z2), x*y*oneMcos - z*s, x*z*oneMcos + y*s, (px*(y2 + z2) - x*(py*y + pz*z))*oneMcos + (py*z - pz*y)*s},
283  {x*y*oneMcos + z*s, y2 + c*(x2 + z2), y*z*oneMcos - x*s, (py*(x2 + z2) - y*(px*x + pz*z))*oneMcos + (pz*x - px*z)*s},
284  {x*z*oneMcos - y*s, y*z*oneMcos + x*s, z2 + c*(x2 + y2), (pz*(x2 + y2) - z*(px*x + py*y))*oneMcos + (px*y - py*x)*s},
285  {0, 0, 0, 1}
286  };

300  // rotation about an arbitrary point (with inverse)
301  template <typename T>
302  void rotmat(SimpleMatrix<T,4,4> *out_mat,
303  SimpleMatrix<T,4,4> *out_inv,
304  T x, T y, T z,
305  T px, T py, T pz,
306  T theta) {
307  T c = cos(theta);
308  T s = sin(theta);
309  T oneMcos = 1 - c;
310  T x2 = x*x;
311  T y2 = y*y;
312  T z2 = z*z;
313  T fac[4][4] = {
314  {x2 + c*(y2 + z2), x*y*oneMcos, x*z*oneMcos, (px*(y2 + z2) - x*(py*y + pz*z))*oneMcos},
315  {x*y*oneMcos, y2 + c*(x2 + z2), y*z*oneMcos, (py*(x2 + z2) - y*(px*x + pz*z))*oneMcos},
316  {x*z*oneMcos, y*z*oneMcos, z2 + c*(x2 + y2), (pz*(x2 + y2) - z*(px*x + py*y))*oneMcos},
317  {0, 0, 0, 1}
318  };
319  T sterms[3][4] = {
320  {0, -z*s, y*s, (py*z - pz*y)*s},
321  {z*s, 0, -x*s, (pz*x - px*z)*s},
322  {-y*s, x*s, 0, (px*y - py*x)*s}
323  };
324 
325  // pre-fill output matrices.
326  T *from = fac[0];
327  T *sfrom = sterms[0];
328  T *mat = out_mat->begin();
329  T *inv = out_inv->begin();
330  std::copy(from, from+16, mat);
331  std::copy(from, from+16, inv);
332 
333  // add sine terms in.
334  for (index_t i = 0; i < 12; i++, mat++, inv++, sfrom++) {
335  *mat += *sfrom;
336  *inv -= *sfrom; // sin(-x) == -sin(x); thus all the s-terms go negative
337  }
338  }

467  template <typename T, index_t N>
468  AffineTransform<T,N> transformation(const SimpleMatrix<T,N,N> &mat) {
469  SimpleMatrix<T,N,N> m_inv;
470  AffineTransform<T,N> atnew;
471 
472  if (N == DYNAMIC_DIM and mat.rows() != mat.cols()) {
473  throw NonsquareMatrixException(mat.rows(), mat.cols());
474  }
475 
476  inv(&m_inv, mat);
477  MatrixRegion region(MatrixCoord::zeros, MatrixCoord(mat.rows()));
478  std::copy( mat.begin(), mat.end(), atnew.mat.region(region).first);
479  std::copy(m_inv.begin(), m_inv.end(), atnew.inv.region(region).first);
480  return atnew;
481  }
