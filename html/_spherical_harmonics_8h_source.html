106 template <typename T>
107 T spherical_harmonic_coeff(index_t l, index_t m, T cos_alt, T azi) {
108  index_t m_a = std::abs(m);
109  static const T sr2 = std::sqrt((T)2);
110  static const T pi = boost::math::constants::pi<T>();
111 
112  T P_lm = legendre(l, m_a, cos_alt);
113 
114  // k contains a sub-expression commonly written: (l-m)!/(l+m)!
115  // which is vulnerable to precision/overflow issues, as very large
116  // intermediate numbers must be stored. here, we compute the reduced
117  // fraction directly:
118  T z = 1;
119  if (l != 0) for (index_t i = l - m_a + 1; i <= l + m_a; i++) { z *= i; }
120 
121  T k = std::sqrt((1/z) * ((2 * l + 1) / (4 * pi)));
122 
123  if (m > 0) {
124  return sr2 * k * std::cos( m * azi) * P_lm;
125  } else if (m < 0) {
126  return sr2 * k * std::sin(-m * azi) * P_lm;
127  } else {
128  return k * legendre(l, 0, cos_alt);
129  }
130 }

307  // protected implementation takes cos(alt), which is just z
308  // for a unit vector. this formulation may better preserve precision when
309  // z is available directly.
310  T _eval(T cos_alt, T azi) const {
311  static const T sr2 = std::sqrt((T)2);
312  static const T pi = boost::math::constants::pi<T>();
313  const index_t l = bands();
314  const T x = cos_alt;
315  T sum = 0;
316  for (index_t m = 0; m < l; m++) {
317  T c_ma = std::cos(m * azi);
318  T s_ma = std::sin(m * azi);
319  T pmm0 = 1;
320  T pmm1 = 1;
321  for (index_t l_i = m; l_i < l; l_i++) {
322  T P_lm = 1;
323 
324  // evaluate P_{l_i}_m(x) from previous values (if extant)
325  if (l_i == m) {
326  if (m > 0) {
327  // compute double factorial of m, with (-1)^m factor
328  for (index_t i = 1; i <= m; i++) {
329  pmm0 *= -(i * 2 - 1);
330  }
331  pmm0 *= std::pow(1 - x * x, m / 2.0);
332  }
333  P_lm *= pmm0;
334  } else if (l_i == m + 1) {
335  pmm1 = x * (2 * m + 1) * pmm0;
336  P_lm *= pmm1;
337  } else {
338  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
339  (l_i - m);
340  pmm0 = pmm1;
341  pmm1 = pmi;
342  P_lm *= pmi;
343  }
344  // P_lm now contains the legendre polynomial value at x
345 
346  // evaluate ((l-m)!/(l+m)!)^-1
347  T z = 1;
348  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
349 
350  // fold Legendre values into SH basis functions
351  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
352  if (m != 0) {
353  sum += P_lm * sr2 * k * c_ma * coeff(l_i, m);
354  sum += P_lm * sr2 * k * s_ma * coeff(l_i, -m);
355  } else {
356  sum += P_lm * k * coeff(l_i, 0);
357  }
358  }
359  }
360  return sum;
361  }

364  void _project(T cos_alt, T azi, T val) {
365  static const T sr2 = std::sqrt((T)2);
366  static const T pi = boost::math::constants::pi<T>();
367  const index_t l = bands();
368  const T x = cos_alt;
369  for (index_t m = 0; m < l; m++) {
370  T c_ma = std::cos(m * azi);
371  T s_ma = std::sin(m * azi);
372  T pmm0 = 1;
373  T pmm1 = 1;
374  for (index_t l_i = m; l_i < l; l_i++) {
375  T P_lm = 1;
376  // evaluate P_{l_i}_m(x) from previous values (if extant)
377  if (l_i == m) {
378  if (m > 0) {
379  for (index_t i = 1; i <= m; i++) {
380  pmm0 *= -(i * 2 - 1);
381  }
382  pmm0 *= std::pow(1 - x * x, m / 2.0);
383  }
384  P_lm = pmm0;
385  } else if (l_i == m + 1) {
386  pmm1 = x * (2 * m + 1) * pmm0;
387  P_lm = pmm1;
388  } else {
389  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
390  (l_i - m);
391  pmm0 = pmm1;
392  pmm1 = pmi;
393  P_lm = pmi;
394  }
395  // P_lm now contains the legendre polynomial value at x
396 
397  // evaluate ((l-m)!/(l+m)!)^-1
398  // this could be done with a LUT for less numerical precision
399  // but I prefer "correct" over "moderately faster".
400  T z = 1;
401  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
402 
403  // fold Legendre values into SH basis functions
404  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
405  if (m != 0) {
406  coeff(l_i, m) += P_lm * sr2 * k * c_ma * val;
407  coeff(l_i, -m) += P_lm * sr2 * k * s_ma * val;
408  } else {
409  coeff(l_i, 0) += P_lm * k * val;
410  }
411  }
412  }

670  void generate_lambertian(index_t order=-1) {
671  // from "On the relationship between radiance and irradiance:
672  // determining the illumination from images of a convex Lambertian
673  // object" by Ramamoorthi and Hanrahan. Note that the coefficients
674  // described therein are those of the spherical harmonics, not zonal
675  // harmonics, so there is an extra factor of sqrt((2l+1)/4pi) applied here.
676  static const T pi = boost::math::constants::pi<T>();
677  const index_t ct = order < 1 ? bands() : std::min(order, bands());
678  clear();
679  T fac = 1;
680  coeffs[0] = 0.25;
681  if (bands() > 1) {
682  coeffs[1] = 0.5;
683  for (index_t n = 2; n < ct; n += 2) { // odd bands are 0.
684  fac *= n * (n - 1);
685  T fac_num = 1;
686  T fac_den = 1;
687  for (index_t i = 1; i <= n / 2; i++) {
688  fac_num *= i + n/2;
689  fac_den *= i;
690  }
691  T k = (2 * n + 1) / (T)2;
692  T A_n = k / (T)((n + 2) * (n - 1));
693  A_n *= std::pow(2, -n) * fac_num / fac_den;
694  if (!(n & (index_t)2)) A_n *= -1;
695  coeffs[n] = A_n;
696  }
697  }

706  void generate_spot(T cos_angle) {
707  // the indefinite integral of the legendre polynomials are
708  // analytic and recurrent. We use these to integrate over the range
709  // of cos_angle for which the spot function is nonzero.
710  static const T pi = boost::math::constants::pi<T>();
711  const index_t n = bands();
712  const T x = std::min((T)1, std::max((T)-1, cos_angle));
713  coeffs[0] = (2 - x - 1) / 2;
714  T p_0 = 1; // P(i - 1, x)
715  T p_1 = x; // P(i, x)
716  T p_i = x; // P(i + 1, x)
717  for (index_t i = 1; i < n; i++) {
718  T k = (2 * i + 1) / ((T)2);
719  p_i = ((2 * i + 1) * x * p_1 - i * p_0) / ((T)i + 1);
720  coeffs[i] = -k * (p_i - p_0) / (2 * i + 1);
721  p_0 = p_1;
722  p_1 = p_i;
723  }
724  }
