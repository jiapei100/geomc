72 T spherical_harmonic_coeff(index_t l, index_t m, T cos_alt, T azi) {
73  index_t m_a = std::abs(m);
74  static const T sr2 = std::sqrt((T)2);
75  static const T pi = boost::math::constants::pi<T>();
76 
77  T P_lm = legendre(l, m_a, cos_alt);
78 
79  // k contains a sub-expression commonly written: (l-m)!/(l+m)!
80  // which is vulnerable to precision/overflow issues, as very large
81  // intermediate numbers must be stored. here, we compute the reduced
82  // fraction directly:
83  T z = 1;
84  if (l != 0) for (index_t i = l - m_a + 1; i <= l + m_a; i++) { z *= i; }
85 
86  T k = std::sqrt((1/z) * ((2 * l + 1) / (4 * pi)));
87 
88  if (m > 0) {
89  return sr2 * k * std::cos( m * azi) * P_lm;
90  } else if (m < 0) {
91  return sr2 * k * std::sin(-m * azi) * P_lm;
92  } else {
93  return k * legendre(l, 0, cos_alt);
94  }
95 }

212  // protected implementation takes cos(alt), which is just z
213  // for a unit vector. this formulation may better preserve precision when
214  // z is available directly.
215  T _eval(T cos_alt, T azi) const {
216  static const T sr2 = std::sqrt((T)2);
217  static const T pi = boost::math::constants::pi<T>();
218  const index_t l = bands();
219  const T x = cos_alt;
220  T sum = 0;
221  for (index_t m = 0; m < l; m++) {
222  T c_ma = std::cos(m * azi);
223  T s_ma = std::sin(m * azi);
224  T pmm0 = 1;
225  T pmm1 = 1;
226  for (index_t l_i = m; l_i < l; l_i++) {
227  T P_lm = 1;
228 
229  // evaluate P_{l_i}_m(x) from previous values (if extant)
230  if (l_i == m) {
231  if (m > 0) {
232  // compute double factorial of m, with (-1)^m factor
233  for (index_t i = 1; i <= m; i++) {
234  pmm0 *= -(i * 2 - 1);
235  }
236  pmm0 *= std::pow(1 - x * x, m / 2.0);
237  }
238  P_lm *= pmm0;
239  } else if (l_i == m + 1) {
240  pmm1 = x * (2 * m + 1) * pmm0;
241  P_lm *= pmm1;
242  } else {
243  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
244  (l_i - m);
245  pmm0 = pmm1;
246  pmm1 = pmi;
247  P_lm *= pmi;
248  }
249  // P_lm now contains the legendre polynomial value at x
250 
251  // evaluate ((l-m)!/(l+m)!)^-1
252  T z = 1;
253  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
254 
255  // fold Legendre values into SH basis functions
256  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
257  if (m != 0) {
258  sum += P_lm * sr2 * k * c_ma * coeff(l_i, m);
259  sum += P_lm * sr2 * k * s_ma * coeff(l_i, -m);
260  } else {
261  sum += P_lm * k * coeff(l_i, 0);
262  }
263  }
264  }
265  return sum;
266  }

269  void _project(T cos_alt, T azi, T val) {
270  static const T sr2 = std::sqrt((T)2);
271  static const T pi = boost::math::constants::pi<T>();
272  const index_t l = bands();
273  const T x = cos_alt;
274  for (index_t m = 0; m < l; m++) {
275  T c_ma = std::cos(m * azi);
276  T s_ma = std::sin(m * azi);
277  T pmm0 = 1;
278  T pmm1 = 1;
279  for (index_t l_i = m; l_i < l; l_i++) {
280  T P_lm = 1;
281  // evaluate P_{l_i}_m(x) from previous values (if extant)
282  if (l_i == m) {
283  if (m > 0) {
284  for (index_t i = 1; i <= m; i++) {
285  pmm0 *= -(i * 2 - 1);
286  }
287  pmm0 *= std::pow(1 - x * x, m / 2.0);
288  }
289  P_lm = pmm0;
290  } else if (l_i == m + 1) {
291  pmm1 = x * (2 * m + 1) * pmm0;
292  P_lm = pmm1;
293  } else {
294  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
295  (l_i - m);
296  pmm0 = pmm1;
297  pmm1 = pmi;
298  P_lm = pmi;
299  }
300  // P_lm now contains the legendre polynomial value at x
301 
302  // evaluate ((l-m)!/(l+m)!)^-1
303  // this could be done with a LUT for less numerical precision
304  // but I prefer "correct" over "moderately faster".
305  T z = 1;
306  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
307 
308  // fold Legendre values into SH basis functions
309  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
310  if (m != 0) {
311  coeff(l_i, m) += P_lm * sr2 * k * c_ma * val;
312  coeff(l_i, -m) += P_lm * sr2 * k * s_ma * val;
313  } else {
314  coeff(l_i, 0) += P_lm * k * val;
315  }
316  }
317  }
