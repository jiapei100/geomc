73 template <typename T>
74 T spherical_harmonic_coeff(index_t l, index_t m, T cos_alt, T azi) {
75  index_t m_a = std::abs(m);
76  static const T sr2 = std::sqrt((T)2);
77  static const T pi = boost::math::constants::pi<T>();
78 
79  T P_lm = legendre(l, m_a, cos_alt);
80 
81  // k contains a sub-expression commonly written: (l-m)!/(l+m)!
82  // which is vulnerable to precision/overflow issues, as very large
83  // intermediate numbers must be stored. here, we compute the reduced
84  // fraction directly:
85  T z = 1;
86  if (l != 0) for (index_t i = l - m_a + 1; i <= l + m_a; i++) { z *= i; }
87 
88  T k = std::sqrt((1/z) * ((2 * l + 1) / (4 * pi)));
89 
90  if (m > 0) {
91  return sr2 * k * std::cos( m * azi) * P_lm;
92  } else if (m < 0) {
93  return sr2 * k * std::sin(-m * azi) * P_lm;
94  } else {
95  return k * legendre(l, 0, cos_alt);
96  }
97 }

272  // protected implementation takes cos(alt), which is just z
273  // for a unit vector. this formulation may better preserve precision when
274  // z is available directly.
275  T _eval(T cos_alt, T azi) const {
276  static const T sr2 = std::sqrt((T)2);
277  static const T pi = boost::math::constants::pi<T>();
278  const index_t l = bands();
279  const T x = cos_alt;
280  T sum = 0;
281  for (index_t m = 0; m < l; m++) {
282  T c_ma = std::cos(m * azi);
283  T s_ma = std::sin(m * azi);
284  T pmm0 = 1;
285  T pmm1 = 1;
286  for (index_t l_i = m; l_i < l; l_i++) {
287  T P_lm = 1;
288 
289  // evaluate P_{l_i}_m(x) from previous values (if extant)
290  if (l_i == m) {
291  if (m > 0) {
292  // compute double factorial of m, with (-1)^m factor
293  for (index_t i = 1; i <= m; i++) {
294  pmm0 *= -(i * 2 - 1);
295  }
296  pmm0 *= std::pow(1 - x * x, m / 2.0);
297  }
298  P_lm *= pmm0;
299  } else if (l_i == m + 1) {
300  pmm1 = x * (2 * m + 1) * pmm0;
301  P_lm *= pmm1;
302  } else {
303  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
304  (l_i - m);
305  pmm0 = pmm1;
306  pmm1 = pmi;
307  P_lm *= pmi;
308  }
309  // P_lm now contains the legendre polynomial value at x
310 
311  // evaluate ((l-m)!/(l+m)!)^-1
312  T z = 1;
313  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
314 
315  // fold Legendre values into SH basis functions
316  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
317  if (m != 0) {
318  sum += P_lm * sr2 * k * c_ma * coeff(l_i, m);
319  sum += P_lm * sr2 * k * s_ma * coeff(l_i, -m);
320  } else {
321  sum += P_lm * k * coeff(l_i, 0);
322  }
323  }
324  }
325  return sum;
326  }

329  void _project(T cos_alt, T azi, T val) {
330  static const T sr2 = std::sqrt((T)2);
331  static const T pi = boost::math::constants::pi<T>();
332  const index_t l = bands();
333  const T x = cos_alt;
334  for (index_t m = 0; m < l; m++) {
335  T c_ma = std::cos(m * azi);
336  T s_ma = std::sin(m * azi);
337  T pmm0 = 1;
338  T pmm1 = 1;
339  for (index_t l_i = m; l_i < l; l_i++) {
340  T P_lm = 1;
341  // evaluate P_{l_i}_m(x) from previous values (if extant)
342  if (l_i == m) {
343  if (m > 0) {
344  for (index_t i = 1; i <= m; i++) {
345  pmm0 *= -(i * 2 - 1);
346  }
347  pmm0 *= std::pow(1 - x * x, m / 2.0);
348  }
349  P_lm = pmm0;
350  } else if (l_i == m + 1) {
351  pmm1 = x * (2 * m + 1) * pmm0;
352  P_lm = pmm1;
353  } else {
354  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) /
355  (l_i - m);
356  pmm0 = pmm1;
357  pmm1 = pmi;
358  P_lm = pmi;
359  }
360  // P_lm now contains the legendre polynomial value at x
361 
362  // evaluate ((l-m)!/(l+m)!)^-1
363  // this could be done with a LUT for less numerical precision
364  // but I prefer "correct" over "moderately faster".
365  T z = 1;
366  if (l_i != 0) for (index_t i = l_i - m + 1; i <= l_i + m; i++) { z *= i; }
367 
368  // fold Legendre values into SH basis functions
369  T k = std::sqrt((1/z) * (2 * l_i + 1) / (4 * pi));
370  if (m != 0) {
371  coeff(l_i, m) += P_lm * sr2 * k * c_ma * val;
372  coeff(l_i, -m) += P_lm * sr2 * k * s_ma * val;
373  } else {
374  coeff(l_i, 0) += P_lm * k * val;
375  }
376  }
377  }

635  void generate_lambertian(index_t order=-1) {
636  // from "On the relationship between radiance and irradiance:
637  // determining the illumination from images of a convex Lambertian
638  // object" by Ramamoorthi and Hanrahan. Note that the coefficients
639  // described therein are those of the spherical harmonics, not zonal
640  // harmonics, so there is an extra factor of sqrt((2l+1)/4pi) applied here.
641  static const T pi = boost::math::constants::pi<T>();
642  const index_t ct = order < 1 ? bands() : std::min(order, bands());
643  clear();
644  T fac = 1;
645  coeffs[0] = 0.25;
646  if (bands() > 1) {
647  coeffs[1] = 0.5;
648  for (index_t n = 2; n < ct; n += 2) { // odd bands are 0.
649  fac *= n * (n - 1);
650  T fac_num = 1;
651  T fac_den = 1;
652  for (index_t i = 1; i <= n / 2; i++) {
653  fac_num *= i + n/2;
654  fac_den *= i;
655  }
656  T k = (2 * n + 1) / (T)2;
657  T A_n = k / (T)((n + 2) * (n - 1));
658  A_n *= std::pow(2, -n) * fac_num / fac_den;
659  if (!(n & (index_t)2)) A_n *= -1;
660  coeffs[n] = A_n;
661  }
662  }
663  }

671  void generate_spot(T cos_angle) {
672  // the indefinite integral of the legendre polynomials are
673  // analytic and recurrent. We use these to integrate over the range
674  // of cos_angle for which the spot function is nonzero.
675  static const T pi = boost::math::constants::pi<T>();
676  const index_t n = bands();
677  const T x = std::min((T)1, std::max((T)-1, cos_angle));
678  coeffs[0] = (2 - x - 1) / 2; // (4 * pi);
679  T p_0 = 1; // P(i - 1, x)
680  T p_1 = x; // P(i, x)
681  T p_i = x; // P(i + 1, x)
682  for (index_t i = 1; i < n; i++) {
683  T k = (2 * i + 1) / ((T)2);
684  p_i = ((2 * i + 1) * x * p_1 - i * p_0) / ((T)i + 1);
685  coeffs[i] = -k * (p_i - p_0) / (2 * i + 1);
686  p_0 = p_1;
687  p_1 = p_i;
688  }
689  }
