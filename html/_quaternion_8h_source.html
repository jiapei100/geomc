102  inline Quat<T> mult(const Quat<T> &q) const {
103  Quat<T> result;
104 
105  const T &x = detail::VecBase<T,4>::x;
106  const T &y = detail::VecBase<T,4>::y;
107  const T &z = detail::VecBase<T,4>::z;
108  const T &w = detail::VecBase<T,4>::w;
109 
110  result.x = w*q.x + x*q.w + y*q.z - z*q.y;
111  result.y = w*q.y - x*q.z + y*q.w + z*q.x;
112  result.z = w*q.z + x*q.y - y*q.x + z*q.w;
113  result.w = w*q.w - x*q.x - y*q.y - z*q.z;
114 
115  return result;
116  }

174  Quat<T> slerp(const Quat<T> &q1, T t) const {
175  // assumed unit quaternion.
176  const Quat<T> &q0 = *this;
177  T dot = q0.dot(q1);
178  if (dot < 0){ //angle is greater than 90. Avoid "long path" rotations.
179  dot = -dot;
180  q1 = -q1;
181  }
182  dot = std::max(0, std::min<T>(1, dot)); //keep <dot> in the domain of acos.
183  T omega = acos(dot);
184  T theta = omega * t;
185  Quat<T> q2 = (q1 - q0 * dot).unit();
186  return cos(theta) * q0 + sin(theta) * q2;
187  }
