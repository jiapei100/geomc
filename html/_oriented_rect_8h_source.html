46  Rect<T,N> bounds() {
47  Vec<T,N> pts[2] = {box.min(), box.max()};
48  Vec<T,N> lo = xf * box.min();
49  Vec<T,N> hi = lo;
50 
51  // for each corner
52  for (index_t i = 1; i < (1 << N); i++) {
53  Vec<T,N> p;
54  // for each coordinate
55  for (index_t j = 0; j < N; j++) {
56  p[j] = pts[(i & (1 << j)) != 0][j];
57  }
58  p = xf * p;
59  lo = std::min(lo, p);
60  hi = std::max(hi, p);
61  }
62 
63  return Rect<T,N>(lo,hi);
64  }

151  bool intersects(const Rect<T,N> &b1) {
152  // here we apply the separating axis theorem.
153  // all edge axes, face normals, and edge-edge normals must be tested.
154  // the first two cases are one and the same; they are the principal
155  // axes of each rect; amounting to 2N tests. The third case is
156  // 9 axes for N=3 and 0 axes for N=2. Other dimensions may be
157  // supported by extending detail::RectAxisHelper to higher N.
158  const static index_t n_corners = 1 << N;
159  Vec<T,N> b0_pts[n_corners];
160  Vec<T,N> b1_pts[n_corners];
161  Vec<T,N> b0_body_extreme[2] = { base_t::box.min(), base_t::box.max() };
162  Vec<T,N> b1_body_extreme[2] = { Vec<T,N>::zeros, b1.max() - b1.min() };
163 
164  // compute world-space points
165  // we put the lower corner of b1 at the origin for simplicity/speed
166  for (index_t i = 0; i < n_corners; i++) {
167  for (index_t j = 0; j < N; j++) {
168  b0_pts[i][j] = b0_body_extreme[(i & (1 << j)) != 0][j];
169  b1_pts[i][j] = b1_body_extreme[(i & (1 << j)) != 0][j];
170  }
171  b0_pts[i] = base_t::xf * b0_pts[i] - b1.min();
172  }
173 
174  // compare along world (i.e. b1's) axes
175  // special case because no projection needed.
176  T lo = std::numeric_limits<T>::max();
177  T hi = std::numeric_limits<T>::lowest();
178  for (index_t i = 0; i < N; i++) {
179  for (index_t j = 0; j < n_corners; j++) {
180  lo = std::min(lo, b0_pts[j][i]);
181  hi = std::max(hi, b0_pts[j][i]);
182  }
183  // if no overlap on this axis, we've found a separating axis.
184  if (lo > b1.max()[i] or hi < b1.min()[i]) return false;
185  }
186 
187  // now test all the remaining axes. For N=2, this is simply
188  // the two principal axes of the oriented box. For N=3, we must also
189  // test each axis which is the cross product of two principal axes,
190  // for up to 12 additional axis tests (we have covered N already).
191  for (detail::RectAxisHelper<T,N> helper(base_t::xf); not helper.done(); helper.next()) {
192  Vec<T,N> axis = helper.axis();
193  if (axis == Vec<T,N>::zeros) continue; // redundant axis
194  Rect<T,1> b0_range, b1_range;
195  b0_range = b1_range = Rect<T,1>(std::numeric_limits<T>::max(), std::numeric_limits<T>::lowest());
196  for (index_t corner = 0; corner < n_corners; corner++) {
197  T b0_proj = axis.dot(b0_pts[corner]);
198  T b1_proj = axis.dot(b1_pts[corner]);
199  b0_range.extendTo(b0_proj);
200  b1_range.extendTo(b1_proj);
201  }
202  // no overlap on this axis; separating axis found.
203  if (not b0_range.intersects(b1_range)) return false;
204  }
205 
206  return true;
207  }

217  bool intersects(const OrientedRect<T,N> &b1) {
218  // we will make ourselves axis-aligned; this is the same as applying
219  // the inverse of our xf. We apply the inverse of xf to b1 too,
220  // to preserve the relationhip between the us. From this, we
221  // fallback to an ORect <-> Rect test.
222  OrientedRect<T,N> b1_in_b0 = b1 / base_t::xf;
223  return b1_in_b0.intersects(base_t::box);
224  }
