47  Rect<T,N> bounds() {
48  Vec<T,N> pts[2] = {box.min(), box.max()};
49  Vec<T,N> lo = xf * box.min();
50  Vec<T,N> hi = lo;
51 
52  // for each corner
53  for (index_t i = 1; i < (1 << N); i++) {
54  Vec<T,N> p;
55  // for each coordinate
56  for (index_t j = 0; j < N; j++) {
57  p[j] = pts[(i & (1 << j)) != 0][j];
58  }
59  p = xf * p;
60  lo = std::min(lo, p);
61  hi = std::max(hi, p);
62  }
63 
64  return Rect<T,N>(lo,hi);
65  }

157  bool intersects(const Rect<T,N> &b1) {
158  // here we apply the separating axis theorem.
159  // all edge axes, face normals, and edge-edge normals must be tested.
160  // the first two cases are one and the same; they are the principal
161  // axes of each rect; amounting to 2N tests. The third case is
162  // 9 axes for N=3 and 0 axes for N=2. Other dimensions may be
163  // supported by extending detail::RectAxisHelper to higher N.
164  const static index_t n_corners = 1 << N;
165  Vec<T,N> b0_pts[n_corners];
166  Vec<T,N> b1_pts[n_corners];
167  Vec<T,N> b0_body_extreme[2] = { base_t::box.min(), base_t::box.max() };
168  Vec<T,N> b1_body_extreme[2] = { b1.min(), b1.max()};
169 
170  // compute world-space points
171  for (index_t c = 0; c < n_corners; c++) {
172  for (index_t i = 0; i < N; i++) {
173  int min_or_max = (c & (1 << i)) != 0;
174  b0_pts[c][i] = b0_body_extreme[min_or_max][i];
175  b1_pts[c][i] = b1_body_extreme[min_or_max][i];
176  }
177  b0_pts[c] = base_t::xf * b0_pts[c];
178  }
179 
180  // compare along world (i.e. b1's) axes
181  // special case because no projection needed.
182  for (index_t i = 0; i < N; i++) {
183  T lo = std::numeric_limits<T>::max();
184  T hi = std::numeric_limits<T>::lowest();
185  for (index_t j = 0; j < n_corners; j++) {
186  lo = std::min(lo, b0_pts[j][i]);
187  hi = std::max(hi, b0_pts[j][i]);
188  }
189  // if no overlap on this axis, we've found a separating axis.
190  if (lo >= b1.max()[i] or hi < b1.min()[i]) return false;
191  }
192 
193  // now test all the remaining axes. For N=2, this is simply
194  // the two principal axes of the oriented box. For N=3, we must also
195  // test each axis which is the cross product of two principal axes,
196  // for up to 12 additional axis tests (we have covered N already).
197  for (detail::RectAxisHelper<T,N> helper(base_t::xf); not helper.done(); helper.next()) {
198  Vec<T,N> axis = helper.axis();
199  if (axis == Vec<T,N>::zeros) continue; // redundant axis
200  Rect<T,1> b0_range, b1_range;
201  b0_range = b1_range = Rect<T,1>(std::numeric_limits<T>::max(), std::numeric_limits<T>::lowest());
202  for (index_t corner = 0; corner < n_corners; corner++) {
203  T b0_proj = axis.dot(b0_pts[corner]);
204  T b1_proj = axis.dot(b1_pts[corner]);
205  b0_range.extendTo(b0_proj);
206  b1_range.extendTo(b1_proj);
207  }
208  // no overlap on this axis; separating axis found.
209  if (not b0_range.intersects(b1_range)) return false;
210  }
211 
212  // overlap on all candidate axes.
213  return true;
214  }

224  bool intersects(const OrientedRect<T,N> &b1) {
225  // we will make ourselves axis-aligned; this is the same as applying
226  // the inverse of our xf. We apply the inverse of xf to b1 too,
227  // to preserve the relationhip between the us. From this, we
228  // fallback to an ORect <-> Rect test.
229  OrientedRect<T,N> b1_in_b0 = b1 / base_t::xf;
230  return b1_in_b0.intersects(base_t::box);
231  }
