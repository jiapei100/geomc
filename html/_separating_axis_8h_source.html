43  Vec<T,N> _getXfAxis(index_t j) {
44  Vec<T,N> a;

71  /********** N=3 Axis Iterator **********/
72 
73  template <typename T>
74  class RectAxisHelper<T,3> : public RectAxisHelperBase<T,3> {
75  private:
76 
77  typedef RectAxisHelperBase<T,3> base_t;
78 
79  public:
80 
81  static const bool SAT_SUPPORTED = true;
82 
83  RectAxisHelper(const AffineTransform<T,3> &xf) : base_t(xf) {}
84 
85  // we don't include the canonical axes, because we handle
86  // them separately for speed. A more user-facing implementation
87  // would iterate over all 15 axes.
88 
89  Vec<T,3> axis() {
90  index_t i = base_t::i;
91  if (i < 3) {
92  // test each of the principal axes in the oriented box
93  return this->_getXfAxis(i);
94  } else { // we are assuming `not done()`
95  // test all combinations of principal axes between
96  // the two boxes. We assume one is axis-aligned.
97  Vec<T,3> b0_axis;
98  Vec<T,3> b1_axis = this->_getXfAxis(i % 3);
99  b0_axis[i / 3] = 1;
100  return b0_axis ^ b1_axis;
101  }
102  }
103 
104  inline bool done() {
105  return base_t::i < 12; // 3 Xf axes, plus 3 * 3 axis combinations.
106  }
107  };
