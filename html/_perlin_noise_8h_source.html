71  T eval(point_t pt) {
72  point_t p0f = std::floor(pt);
73  grid_t p0 = (grid_t)p0f; //grid pt
74  point_t p_modulus = pt - p0f; //pos. within grid
75  const index_t corners = 1 << N;
76  T planeVals[corners];
77  for (index_t c = 0; c < corners; c++) {
78  grid_t cPt;
79  for (index_t axis = 0; axis < N; axis++) {
80  cPt[axis] = (c & (1 << axis)) != 0;
81  }
82  const point_t &grid_grad = get_grid_gradient(p0 + cPt);
83  planeVals[c] = (p_modulus - (point_t)cPt).dot(grid_grad);
84  }
85 
86  for (index_t axis = 0; axis < N; axis++) {
87  T t = fade(p_modulus[axis]); //interpolation parameter
88  for (index_t pair = 0; pair < (1 << (N-axis-1)); pair++) {
89  planeVals[pair] = mix(t, planeVals[pair*2], planeVals[pair*2+1]);
90  }
91  }
92 
93  return planeVals[0];
94  }

107  point_t gradient(point_t pt) {
108  point_t g;
109  //central difference method
110  for (index_t axis = 0; axis < N; axis++) {
111  point_t dx;
112  dx[axis] = PERLIN_EPSILON;
113  g[axis] = eval(pt + dx) - eval(pt - dx);
114  }
115  return g;
116  }

120  inline const point_t& get_grid_gradient(const grid_t &pt) {
121  index_t idx = 0;
122  for (index_t axis = 0; axis < N; axis++) {
123  idx = p[(idx + pt[axis]) & ((index_t)PERLIN_MODULO_MASK)];
124  }
125  return gradients[idx];
126  }
