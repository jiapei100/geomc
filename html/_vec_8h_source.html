151 /* Declare the Vec type that we'll actually use, as a subclass of the hidden base template class.
152  * We do this so that all sizes of vector can share common code, since a template that is merely specialized
153  * does not implicitly inherit functionality from the base template. With this method, Vec<T,2> can be a specialization
154  * of Vec<T,N>, but still keep the functionality of VecBase<T,N>.
155  *
156  * All vectors are sizeof(T)*N and can be packed.
157  */
158 
163 template <typename T, index_t N> class Vec : public detail::VecCommon<T,N> {
164 public:
165 
169  Vec():detail::VecCommon<T,N>(){}
170 
176  Vec(T a):detail::VecCommon<T,N>(a){}
177 
182  Vec(T a[N]):detail::VecCommon<T,N>(a){}

190  Vec(Vec<T,N-1> &v, T a){
191  std::copy(v.begin(), v.end(), detail::VecBase<T,N>::begin());
192  this->get(N-1) = a;
193  }
