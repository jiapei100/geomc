289 // do two matrices share any storage?
290 template <typename Ma, typename Mb>
291 typename boost::enable_if_c<
292  detail::IsMatrix<Ma>::val and detail::IsMatrix<Mb>::val,
293  bool>::type
294 mtx_aliases_storage(const Ma &a, const Mb &b) {
295  // todo: currently assumes no /internal/ aliasing.
296  typename Ma::storagebuffer_t buf_a = a.getStorageIDBuffer();
297  typename Mb::storagebuffer_t buf_b = b.getStorageIDBuffer();
298 
299  a.getStorageIDs(buf_a.get()); // may be dynamically allocated
300  b.getStorageIDs(buf_b.get()); // but usually static, and usually only one ptr
301  for (index_t i = 0; i < a.getStorageIDCount(); i++) {
302  for (index_t j = 0; j < b.getStorageIDCount(); j++) {
303  if (buf_a.get()[i] == buf_b.get()[j]) return true;
304  }
305  }
306 
307  return false;
308 }

421 // (a x b) * (b x c) -> (a x c)
422 /* Implementation:
423  *
424  * This function's main activity is to guarantee dimension safety and
425  * protect against memory aliasing. The actual calculation is performed
426  * by _ImplMatrixMul, for which there are specializations over meaningful
427  * operand types. _ImplMatrixMul also provides a resultant type (usually
428  * a SimpleMatrix) for use when a temporary destination buffer is needed.
429  *
430  * Because Vecs do not have Matrix-like 2D dimension, an _ImplMtxAdaptor
431  * is used to make Vecs interoperable with matrices. The adaptor must be
432  * told whether a vector is expected to be a row vector or a column vector.
433  * Result vectors may be either a row or column vector depending on the
434  * order of the operands, so an _ImplVecMulOrient is used to select the
435  * appropriate orientation for result vectors.
436  */
437 template <typename Md, typename Ma, typename Mb>
438 typename boost::enable_if_c<
439  detail::MatrixMultipliable<Ma,Mb>::val and
440  detail::_ImplMtxResult<Ma, Mb, Md>::agreement == detail::MTX_RESULT_MATCH,
441  Md&>::type
442 mul(Md *into, const Ma &a, const Mb &b) {
443  // adaptor types for each argument:
444  typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
445  typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
446  typedef detail::_ImplMtxAdaptor<Md, detail::_ImplVecMulOrient<Ma,Mb,Md>::orient> D;
447  // multiplier implementation:
448  typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
449  typedef typename mult_t::return_t buffer_t;
450 
451  #ifdef GEOMC_MTX_CHECK_DIMS
452  // do the source matrix dimensions agree?
453  // any compiler worth its salt should eliminate this test for template instantiations with static-dimensioned operands
454  if ((A::COLDIM == DYNAMIC_DIM or B::ROWDIM == DYNAMIC_DIM) and A::cols(a) != B::rows(b)) {
455  throw DimensionMismatchException(A::rows(a), A::cols(a), B::rows(b), B::cols(b));
456  }
457  // does the destination matrix have correct dims?
458  // again, runtime checks are to be avoided.
459  if (((D::ROWDIM == DYNAMIC_DIM or A::ROWDIM == DYNAMIC_DIM) and A::rows(a) != D::rows(*into)) or
460  ((D::COLDIM == DYNAMIC_DIM or B::COLDIM == DYNAMIC_DIM) and B::cols(b) != D::cols(*into))) {
461  throw DimensionMismatchException(D::rows(*into), D::cols(*into), A::rows(a), B::cols(b));
462  }
463  #endif
464 
465  #ifdef GEOMC_MTX_CHECK_ALIASING
466  // allocate a temp destination matrix, in case of storage aliasing
467  if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
468  buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
469  mult_t::mul(&tmp, a, b);
470  detail::_mtxcopy(into, tmp);
471  } else {
472  mult_t::mul(into, a, b);
473  }
474  #else
475  mult_t::mul(into, a, b);
476  #endif
477 
478  return *into;
479 }

481 // vec <- mtx * mtx
482 // Special case where matrices have dynamic size and destination is a vector.
483 // The desired row/col orientation of the vector cannot be determined at compile
484 // time, so special logic is needed to check both configurations.
485 template <typename Md, typename Ma, typename Mb>
486 typename boost::enable_if_c<
487  detail::MatrixMultipliable<Ma,Mb>::val and
488  detail::_ImplMtxResult<Ma,Mb,Md>::agreement == detail::MTX_RESULT_UNKNOWN,
489  Md&>::type
490 mul(Md *into, const Ma &a, const Mb &b) {
491  // adaptors types for arguments:
492  typedef detail::_ImplMtxAdaptor<Ma, detail::ORIENT_VEC_ROW> A;
493  typedef detail::_ImplMtxAdaptor<Mb, detail::ORIENT_VEC_COL> B;
494  //multiplier implementation
495  typedef detail::_ImplMtxMul<Ma,Mb> mult_t;
496  typedef typename mult_t::return_t buffer_t;
497 
498  #ifdef GEOMC_MTX_CHECK_DIMS
499  // trial orientation adaptors:
500  typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_ROW> Dr;
501  typedef detail::_ImplMtxAdaptor<Md, detail::ORIENT_VEC_COL> Dc;
502 
503  // if neither configuration yields dimension agreement:
504  if (not ((Dr::rows(*into) == A::rows(a) and Dr::cols(*into) == B::cols(b)) or
505  (Dc::rows(*into) == A::rows(a) and Dc::cols(*into) == B::cols(b))) ) {
506  throw DimensionMismatchException(Dr::rows(*into), Dr::cols(*into), A::rows(a), B::cols(b));
507  }
508  #endif
509 
510  #ifdef GEOMC_MTX_CHECK_ALIASING
511  if (mtx_aliases_storage(*into, a) or mtx_aliases_storage(*into, b)) {
512  buffer_t tmp = detail::_ImplMtxInstance<buffer_t>::instance(A::rows(a), B::cols(b));
513  mult_t::mul(&tmp, a, b);
514  detail::_mtxcopy(into, tmp);
515  } else {
516  mult_t::mul(into, a, b);
517  }
518  #else
519  mult_t::mul(into, a, b);
520  #endif
521 
522  return *into;
523 }

611  }

614 }

656 template <typename Md, typename Mx>
657 bool inv(Md *into, const Mx &src,
658  M_ENABLE_IF_C(
659  (detail::MatrixDimensionMatch<Md,Mx>::isStaticMatch and
660  (Mx::ROWDIM == Mx::COLDIM or Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM))
661  )) {
662  typedef detail::_ImplMtxInv<Mx> inv_t;
663 #ifdef GEOMC_MTX_CHECK_DIMS
664  if (Mx::ROWDIM * Mx::COLDIM == DYNAMIC_DIM and src.rows() != src.cols()) {
665  throw NonsquareMatrixException(src.rows(), src.cols());
666  }
667  detail::MatrixDimensionMatch<Md,Mx>::check(*into, src);
668 #endif
669  // matrix inv() implementations, unlike mul() or txpose(), shall assume
670  // that the destination and source may be aliased. this design decision
671  // was made because several inv() implementations are agnostic about
672  // aliasing (they perform internal copies), making a check unnecessary.
673  // therefore we delegate the aliasing check to those inv()s which care.
674  return inv_t::inv(into, src);
675 }
