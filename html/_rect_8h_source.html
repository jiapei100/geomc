453  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
454  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
455  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
456  index_t near_axis = 0;
457  index_t far_axis = 0;
458  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
459  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
460  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
461 
462  // test ray intersection with an infinite slab along each axis
463  // box intersections are the farthest near hit and nearest far hit
464  for (index_t axis = 0; axis < N; axis++) {
465  if (r.direction[axis] == 0) {
466  // ray direction tangent to test planes, no intersection along this axis
467  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
468  // origin outside of test slab; miss
469  return hit;
470  }
471  } else {
472  // coordinate of hit, along tested axis
473  T c1 = maxs[axis];
474  T c2 = mins[axis];
475  // ray multiple of hit
476  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
477  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
478 
479  // order s1, s2 to (near, far) (aka front, back)
480  if (s2 < s1) {
481  std::swap(s1,s2);
482  std::swap(c1,c2);
483  }
484  if (s1 > near_root) {
485  near_root = s1;
486  near_axis = axis;
487  near_coord = c1;
488  }
489  if (s2 < far_root) {
490  far_root = s2;
491  far_axis = axis;
492  far_coord = c2;
493  }
494  }
495  }
496 
497  if (near_root > far_root) {
498  // miss
499  return hit;
500  } else if (near_root > 0 and (sides & HIT_FRONT)) {
501  // hit front
502  hit.p = r * near_root;
503  hit.p[near_axis] = near_coord;
504  hit.s = near_root;
505  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
506  hit.side = HIT_FRONT;
507  hit.hit = true;
508  } else if (far_root > 0 and (sides & HIT_BACK)) {
509  // hit back
510  hit.p = r * far_root;
511  hit.p[far_axis] = far_coord;
512  hit.s = far_root;
513  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
514  hit.side = HIT_BACK;
515  hit.hit = true;
516  }
517  // if encountered cases above; hit.
518  // else miss (box entirely behind ray origin)
519  return hit;
520  }
