2  * Rect.h
3  *
4  * An N-dimensional axis-aligned box.
5  *
6  * For N > 1, point_t is Vec<T,N>. For N=1, point_t is T;
7  * thus a Rect<T,1> may be used to naturally represent a 1D range.
8  *
9  * Example: Rect<double,4>::point_t is Vec<double,4>; getMin() returns Vec4d.
10  * Rect<double,1>::point_t is double; getMin() returns a double.
11  *
12  * If you wanted to experiment with interesting types of bound (like T=Expr, double[], function, ...),
13  * then you could use Rect<MyWeirdType, 1> (this is equivalent to the original RectBound implementation).
14  *

16  * Author: Tim Babb

258  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
259  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
260  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
261  index_t near_axis = 0;
262  index_t far_axis = 0;
263  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
264  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
265  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
266 
267  // test ray intersection with an infinite slab along each axis
268  // box intersections are the farthest near hit and nearest far hit
269  for (index_t axis = 0; axis < N; axis++){
270  if (r.direction[axis] == 0){
271  // ray direction tangent to test planes, no intersection along this axis
272  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
273  // origin outside of test slab; miss
274  return hit;
275  }
276  } else {
277  // coordinate of hit, along tested axis
278  T c1 = maxs[axis];
279  T c2 = mins[axis];
280  // ray multiple of hit
281  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
282  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
283 
284  // order s1, s2 to (near, far) (aka front, back)
285  if (s2 < s1){
286  std::swap(s1,s2);
287  std::swap(c1,c2);
288  }
289  if (s1 > near_root){
290  near_root = s1;
291  near_axis = axis;
292  near_coord = c1;
293  }
294  if (s2 < far_root){
295  far_root = s2;
296  far_axis = axis;
297  far_coord = c2;
298  }
299  }
300  }
301 
302  if (near_root > far_root){
303  // miss
304  return hit;
305  } else if (near_root > 0 and (sides & HIT_FRONT)){
306  // hit front
307  hit.p = r * near_root;
308  hit.p[near_axis] = near_coord;
309  hit.s = near_root;
310  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
311  hit.side = HIT_FRONT;
312  hit.hit = true;
313  } else if (far_root > 0 and (sides & HIT_BACK)) {
314  // hit back
315  hit.p = r * far_root;
316  hit.p[far_axis] = far_coord;
317  hit.s = far_root;
318  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
319  hit.side = HIT_BACK;
320  hit.hit = true;
321  }
322  // if encountered cases above; hit.
323  // else miss (box entirely behind ray origin)
324  return hit;
325  }
