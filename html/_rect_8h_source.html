454  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
455  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
456  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
457  index_t near_axis = 0;
458  index_t far_axis = 0;
459  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
460  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
461  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
462 
463  // test ray intersection with an infinite slab along each axis
464  // box intersections are the farthest near hit and nearest far hit
465  for (index_t axis = 0; axis < N; axis++) {
466  if (r.direction[axis] == 0) {
467  // ray direction tangent to test planes, no intersection along this axis
468  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
469  // origin outside of test slab; miss
470  return hit;
471  }
472  } else {
473  // coordinate of hit, along tested axis
474  T c1 = maxs[axis];
475  T c2 = mins[axis];
476  // ray multiple of hit
477  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
478  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
479 
480  // order s1, s2 to (near, far) (aka front, back)
481  if (s2 < s1) {
482  std::swap(s1,s2);
483  std::swap(c1,c2);
484  }
485  if (s1 > near_root) {
486  near_root = s1;
487  near_axis = axis;
488  near_coord = c1;
489  }
490  if (s2 < far_root) {
491  far_root = s2;
492  far_axis = axis;
493  far_coord = c2;
494  }
495  }
496  }
497 
498  if (near_root > far_root) {
499  // miss
500  return hit;
501  } else if (near_root > 0 and (sides & HIT_FRONT)) {
502  // hit front
503  hit.p = r * near_root;
504  hit.p[near_axis] = near_coord;
505  hit.s = near_root;
506  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
507  hit.side = HIT_FRONT;
508  hit.hit = true;
509  } else if (far_root > 0 and (sides & HIT_BACK)) {
510  // hit back
511  hit.p = r * far_root;
512  hit.p[far_axis] = far_coord;
513  hit.s = far_root;
514  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
515  hit.side = HIT_BACK;
516  hit.hit = true;
517  }
518  // if encountered cases above; hit.
519  // else miss (box entirely behind ray origin)
520  return hit;
521  }
