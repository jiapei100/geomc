466  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
467  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
468  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
469  index_t near_axis = 0;
470  index_t far_axis = 0;
471  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
472  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
473  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
474 
475  // test ray intersection with an infinite slab along each axis
476  // box intersections are the farthest near hit and nearest far hit
477  for (index_t axis = 0; axis < N; axis++) {
478  if (r.direction[axis] == 0) {
479  // ray direction tangent to test planes, no intersection along this axis
480  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
481  // origin outside of test slab; miss
482  return hit;
483  }
484  } else {
485  // coordinate of hit, along tested axis
486  T c1 = maxs[axis];
487  T c2 = mins[axis];
488  // ray multiple of hit
489  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
490  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
491 
492  // order s1, s2 to (near, far) (aka front, back)
493  if (s2 < s1) {
494  std::swap(s1,s2);
495  std::swap(c1,c2);
496  }
497  if (s1 > near_root) {
498  near_root = s1;
499  near_axis = axis;
500  near_coord = c1;
501  }
502  if (s2 < far_root) {
503  far_root = s2;
504  far_axis = axis;
505  far_coord = c2;
506  }
507  }
508  }
509 
510  if (near_root > far_root) {
511  // miss
512  return hit;
513  } else if (near_root > 0 and (sides & HIT_FRONT)) {
514  // hit front
515  hit.p = r * near_root;
516  hit.p[near_axis] = near_coord;
517  hit.s = near_root;
518  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
519  hit.side = HIT_FRONT;
520  hit.hit = true;
521  } else if (far_root > 0 and (sides & HIT_BACK)) {
522  // hit back
523  hit.p = r * far_root;
524  hit.p[far_axis] = far_coord;
525  hit.s = far_root;
526  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
527  hit.side = HIT_BACK;
528  hit.hit = true;
529  }
530  // if encountered cases above; hit.
531  // else miss (box entirely behind ray origin)
532  return hit;
533  }
