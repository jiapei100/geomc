467  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
468  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
469  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
470  index_t near_axis = 0;
471  index_t far_axis = 0;
472  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
473  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
474  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
475 
476  // test ray intersection with an infinite slab along each axis
477  // box intersections are the farthest near hit and nearest far hit
478  for (index_t axis = 0; axis < N; axis++) {
479  if (r.direction[axis] == 0) {
480  // ray direction tangent to test planes, no intersection along this axis
481  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
482  // origin outside of test slab; miss
483  return hit;
484  }
485  } else {
486  // coordinate of hit, along tested axis
487  T c1 = maxs[axis];
488  T c2 = mins[axis];
489  // ray multiple of hit
490  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
491  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
492 
493  // order s1, s2 to (near, far) (aka front, back)
494  if (s2 < s1) {
495  std::swap(s1,s2);
496  std::swap(c1,c2);
497  }
498  if (s1 > near_root) {
499  near_root = s1;
500  near_axis = axis;
501  near_coord = c1;
502  }
503  if (s2 < far_root) {
504  far_root = s2;
505  far_axis = axis;
506  far_coord = c2;
507  }
508  }
509  }
510 
511  if (near_root > far_root) {
512  // miss
513  return hit;
514  } else if (near_root > 0 and (sides & HIT_FRONT)) {
515  // hit front
516  hit.p = r * near_root;
517  hit.p[near_axis] = near_coord;
518  hit.s = near_root;
519  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
520  hit.side = HIT_FRONT;
521  hit.hit = true;
522  } else if (far_root > 0 and (sides & HIT_BACK)) {
523  // hit back
524  hit.p = r * far_root;
525  hit.p[far_axis] = far_coord;
526  hit.s = far_root;
527  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
528  hit.side = HIT_BACK;
529  hit.hit = true;
530  }
531  // if encountered cases above; hit.
532  // else miss (box entirely behind ray origin)
533  return hit;
534  }
