429  Hit<T,N> trace(const Ray<T,N> &r, HitSide sides) const {
430  T near_root = std::numeric_limits<T>::lowest(); // non-denormal extremes, for speed (as opposed to inf).
431  T far_root = std::numeric_limits<T>::max(); // besides, infinity may not be defined for T.
432  index_t near_axis = 0;
433  index_t far_axis = 0;
434  T near_coord = 0; // for guaranteeing that the hit point is exactly on the rect surface,
435  T far_coord = 0; // regardless of precision and rounding error in the ray arithmetic.
436  Hit<T,N> hit = Hit<T,N>(r, sides); // defaults to miss
437 
438  // test ray intersection with an infinite slab along each axis
439  // box intersections are the farthest near hit and nearest far hit
440  for (index_t axis = 0; axis < N; axis++) {
441  if (r.direction[axis] == 0) {
442  // ray direction tangent to test planes, no intersection along this axis
443  if (r.origin[axis] < mins[axis] || r.origin[axis] > maxs[axis]) {
444  // origin outside of test slab; miss
445  return hit;
446  }
447  } else {
448  // coordinate of hit, along tested axis
449  T c1 = maxs[axis];
450  T c2 = mins[axis];
451  // ray multiple of hit
452  T s1 = (c1 - r.origin[axis]) / r.direction[axis];
453  T s2 = (c2 - r.origin[axis]) / r.direction[axis];
454 
455  // order s1, s2 to (near, far) (aka front, back)
456  if (s2 < s1) {
457  std::swap(s1,s2);
458  std::swap(c1,c2);
459  }
460  if (s1 > near_root) {
461  near_root = s1;
462  near_axis = axis;
463  near_coord = c1;
464  }
465  if (s2 < far_root) {
466  far_root = s2;
467  far_axis = axis;
468  far_coord = c2;
469  }
470  }
471  }
472 
473  if (near_root > far_root) {
474  // miss
475  return hit;
476  } else if (near_root > 0 and (sides & HIT_FRONT)) {
477  // hit front
478  hit.p = r * near_root;
479  hit.p[near_axis] = near_coord;
480  hit.s = near_root;
481  hit.n[near_axis] = -copysign(1, r.direction[near_axis]);
482  hit.side = HIT_FRONT;
483  hit.hit = true;
484  } else if (far_root > 0 and (sides & HIT_BACK)) {
485  // hit back
486  hit.p = r * far_root;
487  hit.p[far_axis] = far_coord;
488  hit.s = far_root;
489  hit.n[far_axis] = copysign(1, r.direction[far_axis]);
490  hit.side = HIT_BACK;
491  hit.hit = true;
492  }
493  // if encountered cases above; hit.
494  // else miss (box entirely behind ray origin)
495  return hit;
496  }
