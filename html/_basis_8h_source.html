26 // makes use of recurrent definition of legendre polys to save re-execs
27 // evaluate a column m and -m of Legendre polynomials and stuff them
28 // into the appropriate place in the SphericalHarmonics' coeff table
29 template <typename T, index_t N>
30 void legendre(SphericalHarmonics<T,N> *sh, index_t m, T x) {
31  index_t l = sh->bands();
32  T pmm0 = 1;
33  if (m > 0) {
34  // compute double factorial of 2m - 1
35  // fold in the (-1)^m
36  for (index_t i = 1; i <= m; i++) {
37  pmm0 *= -(i * 2 - 1);
38  }
39  pmm0 *= std::pow(1 - x * x, m / 2.0);
40  }
41  sh->coeff(m,m) = sh->coeff(m,-m) = pmm0;
42  if (m == l - 1) return; // there is no higher band
43  T pmm1 = sh->coeff(m+1,m) = sh->coeff(m+1,-m) = x * (2 * m + 1) * pmm0;
44  if (m == l - 2) return; // there is no higher band
45  for (index_t l_i = m + 2; l_i < l; l_i++) {
46  T pmi = (x * (2 * l_i - 1) * pmm1 - (l_i + m - 1) * pmm0) / (l_i - m);
47  pmm0 = pmm1;
48  pmm1 = sh->coeff(l_i,m) = sh->coeff(l_i,-m) = pmi;
49  }
50 }

60 template <typename T>
61 T legendre(index_t l, index_t m, T x) {
62  T pmm0 = 1;
63  if (m > 0) {
64  // compute double factorial of 2m - 1
65  // fold in the (-1)^m
66  for (index_t i = 1; i <= m; i++) {
67  pmm0 *= -(i * 2 - 1);
68  }
69  pmm0 *= std::pow(1 - x * x, m / 2.0);
70  }
71  if (l == m) return pmm0;
72  T pmm1 = x * (2 * m + 1) * pmm0;
73  if (l == m + 1) return pmm1;
74  for (index_t i = m + 2; i <= l; i++) {
75  T pmi = (x * (2 * i - 1) * pmm1 - (i + m - 1) * pmm0) / (i - m);
76  pmm0 = pmm1;
77  pmm1 = pmi;
78  }
79  return pmm1;
80 }
