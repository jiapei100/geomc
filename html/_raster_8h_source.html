255  sample_t sample(const coord_t &pt, EdgeBehavior edge=EDGE_CLAMP, Interpolation interp=INTERP_LINEAR) const {
256  grid_t gridPt;
257  pt = toGridSpace(pt);
258 
259  switch (interp) {
260  case INTERP_NEAREST:
261  gridPt = grid_t(pt + coord_t(0.5));
262  return sample_discrete(gridPt, edge);
263  case INTERP_LINEAR:
264  {
265  sample_t buf_l[1<<M];
266  gridPt = (grid_t)pt;
267  coord_t s = pt - ((coord_t)gridPt);
268 
269  // copy surrounding 2^M sample pts into a contiguous buffer
270  copy(buf_l, Rect<index_t,M>(gridPt, gridPt + grid_t(2)), edge);
271 
272  return interp_linear(s.begin(), buf_l, M);
273  }
274  case INTERP_CUBIC:
275  {
276  sample_t buf_c[1<<(2*M)];
277  gridPt = (grid_t)pt;
278  coord_t s = pt - ((coord_t)gridPt);
279 
280  // copy surrounding 4^M sample pts into a contiguous buffer
281  copy(buf_c, Rect<index_t,M>(gridPt - grid_t(1), gridPt + grid_t(3)), edge);
282 
283  return interp_cubic(s.begin(), buf_c, M);
284  }
285  default:
286  return m_abyss;
287  }
288  }
