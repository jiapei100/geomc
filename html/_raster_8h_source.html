174  sample_t sample(const coord_t &pt, EdgeBehavior edge=EDGE_CLAMP, Interpolation interp=INTERP_LINEAR) const {
175  grid_t gridPt;
176 
177  switch (interp){
178  case INTERP_NEAREST:
179  gridPt = grid_t(pt + coord_t(0.5));
180  return sample_discrete(gridPt, edge);
181  case INTERP_LINEAR:
182  {
183  sample_t buf_l[1<<M];
184  gridPt = (grid_t)pt;
185  coord_t s = pt - ((coord_t)gridPt);
186 
187  // copy surrounding 2^M sample pts into a contiguous buffer
188  copy(buf_l, Rect<index_t,M>(gridPt, gridPt + grid_t(2)), edge);
189 
190  return interp_linear(s.begin(), buf_l, M);
191  }
192  case INTERP_CUBIC:
193  {
194  sample_t buf_c[1<<(2*M)];
195  gridPt = (grid_t)pt;
196  coord_t s = pt - ((coord_t)gridPt);
197 
198  // copy surrounding 4^M sample pts into a contiguous buffer
199  copy(buf_c, Rect<index_t,M>(gridPt - grid_t(1), gridPt + grid_t(3)), edge);
200 
201  return interp_cubic(s.begin(), buf_c, M);
202  }
203  default:
204  return abyss;
205  }
206  }
