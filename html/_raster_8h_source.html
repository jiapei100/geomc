168  sample_t sample(const coord_t &pt, EdgeBehavior edge=EDGE_CLAMP, Interpolation interp=INTERP_LINEAR) const {
169  grid_t gridPt;
170 
171  switch (interp){
172  case INTERP_NEAREST:
173  gridPt = grid_t(pt + coord_t(0.5));
174  return sample_discrete(gridPt, edge);
175  case INTERP_LINEAR:
176  {
177  sample_t buf_l[1<<M];
178  gridPt = (grid_t)pt;
179  coord_t s = pt - ((coord_t)gridPt);
180 
181  // copy surrounding 2^M sample pts into a contiguous buffer
182  copy(buf_l, Rect<index_t,M>(gridPt, gridPt + grid_t(2)), edge);
183 
184  return interp_linear(s.begin(), buf_l, M);
185  }
186  case INTERP_CUBIC:
187  {
188  sample_t buf_c[1<<(2*M)];
189  gridPt = (grid_t)pt;
190  coord_t s = pt - ((coord_t)gridPt);
191 
192  // copy surrounding 4^M sample pts into a contiguous buffer
193  copy(buf_c, Rect<index_t,M>(gridPt - grid_t(1), gridPt + grid_t(3)), edge);
194 
195  return interp_cubic(s.begin(), buf_c, M);
196  }
197  default:
198  return abyss;
199  }
200  }

262  bool contains(const grid_t &pt) const {

265  index_t *p_i = grid_info::iterator(pt);
