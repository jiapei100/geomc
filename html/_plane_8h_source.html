23  // plane equation: ax + by + cz + d = 0
24  // normal = (a,b,c)
25  // d = distance along normal from origin to plane
26  // negative means move forward along normal,
27  // since we will have to go "more positive"

68  // returns an s such that p - N*s is on the plane,
69  // or what multiple of the normal must I travel off the plane?
70  // in other words, positive "above" the plane (in the
71  // direction of the normal), and negative "below".
72  inline T distance(const Vec<T,N> &p) const {
73  return normal.dot(p) + d;
74  }
75 
76  inline bool contains(const Vec<T,N> &p) const {
77  return distance(p) < 0;
78  }

80  inline Vec<T,N> project(const Vec<T,N> &p) const {
81  return -(p.dot(normal) + d) * normal + p;
82  }

85  inline Vec<T,N> origin() const {

89  void apply(const AffineTransform<T,N> &xf) {
90  Vec<T,N> p0 = normal*d; // a point on the plane
91  Vec<T,N> p1 = xf.apply(p0); // the new position of that point
92  normal = xf.applyNormal(normal).unit(); // construct a plane with transformed normal
93  d = p1.dot(normal); // and transformed position
94  }
95 
96  void applyInverse(const AffineTransform<T,N> &xf) {
97  Vec<T,N> p0 = normal*d;
98  Vec<T,N> p1 = xf.applyInverse(p0);
99  normal = xf.applyInverseNormal(normal).unit();
100  d = p1.dot(normal);
101  }

103  inline bool intersects(const Vec<T,N> *hull, index_t npts) const {
104  bool negative;
105  for (index_t i = 0; i < npts; i++){
106  T dist = distance(hull[i]);
107  if (i > 0){
108  // if a point lies on the plane, or another point lies on the
109  // other side of the plane, then this shape crosses the plane.
110  if (dist == 0 || ((dist < 0) != negative)){
111  return false;
112  }
113  } else {
114  negative = dist < 0;
115  }
116  }
117  return false;
118  }
